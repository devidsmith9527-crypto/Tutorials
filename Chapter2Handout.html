namespace OopMvcApplication.Models
{
    public class Product
    {
        public int Id { get; set; }
        public string Name { get; set; } = string.Empty;
        public decimal Price { get; set; }
        public int Qty { get; set; }

        public Product(int id, string name, decimal price, int qty)
        {
            Id = id; Name = name; Price = price; Qty = qty;
        }

        public decimal CalculateTotal() => Price * Qty;
    }
}
Controllers/ProductController.cs
using Microsoft.AspNetCore.Mvc;

{

        {
            List<Product> products = new()

                new Product(1, "iPhone 15 Pro", 999.00m, 2),
                new Product(2, "MacBook Pro M3", 1999.00m, 1)
            };
            return View(products);
        }
    }
}

១. សំណួរ ខុស/ត្រូវ (True/False)

សេចក្តីណែនាំ៖ សូមគូសសញ្ញា (ត្រូវ) ប្រសិនបើប្រយោគខាងក្រោមត្រឹមត្រូវ ឬសញ្ញា (ខុស) ប្រសិនបើខុស។

  1. C# គឺជាភាសាប្រភេទ Strongly-typed Language ដែលអថេរទាំងអស់ត្រូវតែប្រកាសប្រភេទទិន្នន័យជានិច្ច។
  2. អថេរប្រភេទ int អាចផ្ទុកទិន្នន័យជាតម្លៃទសភាគ (Decimal) បានយ៉ាងរលូន។
  3. ប្រភេទទិន្នន័យ decimal ត្រូវបានណែនាំឱ្យប្រើប្រាស់សម្រាប់ការគណនាហិរញ្ញវត្ថុ ឬលុយកាក់ដើម្បីជៀសវាងបញ្ហាបង្គត់លេខ។
  4. ViewBag ប្រើប្រាស់យន្តការ Dictionary ប្រភេទ Key-Value ដែលតម្រូវឱ្យធ្វើការ Cast ប្រភេទជានិច្ច។
  5. លក្ខខណ្ឌ @if នៅក្នុង Razor Syntax ត្រូវបានដំណើរការនៅលើ Server មុននឹងបញ្ជូនកូដ HTML ទៅកាន់ Client Browser។
  6. រង្វិលជុំ @foreach មិនតម្រូវឱ្យយើងប្រើប្រាស់សន្ទស្សន៍ (Index) ដើម្បីអានសមាជិក List ឡើយ។
  7. Class គឺជាវត្ថុពិតជាក់ស្តែងដែលកំពុងដំណើរការ និងប្រើប្រាស់ Memory នៅក្នុងកុំព្យូទ័រ។
  8. Property set ត្រូវបានប្រើប្រាស់ដើម្បីអនុញ្ញាតឱ្យ Client កែប្រែ ឬបញ្ចូលតម្លៃទៅក្នុងអថេររបស់ Class។
  9. Constructor គឺជា Method ពិសេសរបស់ Class ដំណើរការដំបូងគេបង្អស់ពេលបង្កើត new Object។
  10. Strongly-typed View ជួយការពារការសរសេរកូដខុសប្រភេទ Data Type ក្នុងពេល Compile-time។

២. សំណួរ ពហុជ្រើសរើស (MCQs)

    • A. int
    • B. string
    • C. decimal
    • D. bool

    • D. do-while

    • A. read
    • B. write
    • C. set
    • D. get
  6. តើមួយណាជាការប្រៀបធៀប Class និង Object បានត្រឹមត្រូវ?

    • B. Class ជាប្លង់គំរូ, Object ជាផលិតផលសម្រេចក្នុង Memory
    • C. ទាំងពីរជាប្លង់ដូចគ្នា
    • D. ទាំងពីរជាផលិតផលដូចគ្នា
  7. តើ Constructor ដំណើរការនៅពេលណា?
    • A. ដំណើរការនៅពេលហៅប្រើប្រាស់ Method
    • B. ដំណើរការដំបូងបង្អស់នៅពេល Object ត្រូវបានបង្កើត (new)
    • C. ដំណើរការនៅពេលលុប Object ចេញពី Memory
    • D. មិនដំណើរការដោយស្វ័យប្រវត្តឡើយ

ក. កំរិតស្រាល (Easy Level) - ២ លំហាត់
  • លំហាត់ 5.1 (ContactInfo): ចូរសរសេរ Controller Action មួយឈ្មោះ ContactInfo() ដើម្បីបញ្ជូនព័ត៌មានទំនាក់ទំនងរបស់សាកលវិទ្យាល័យរួមមាន Email, Phone, Telegram ទៅកាន់ View តាមរយៈ ViewBag រួចបង្ហាញព័ត៌មានទាំងនោះក្នុងទម្រង់ HTML card ស្អាត។
  • លំហាត់ 5.2 (CheckGrade): ចូរសរសេរ Controller Action មួយដែលទទួលបានពិន្ទុ (double score) រួចបញ្ជូនពិន្ទុទៅកាន់ View។ នៅក្នុង View ត្រូវប្រើប្រាស់ Razor @if-else ដើម្បីពិនិត្យលក្ខខណ្ឌ៖ ប្រសិនបើពិន្ទុចាប់ពី ៥០ ឡើងទៅ ត្រូវបង្ហាញសារពណ៌បៃតង "អ្នកបានប្រឡងជាប់" បើផ្ទុយទៅវិញ បង្ហាញសារពណ៌ក្រហម "អ្នកបានប្រឡងធ្លាក់"។
ខ. កំរិតមធ្យម (Medium Level) - ២ លំហាត់

public class StudentScore {
    public int Id { get; set; }
    public string Name { get; set; } = string.Empty;
    public double Attendance { get; set; }
    public double Midterm { get; set; }
    public double Final { get; set; }
    public StudentScore(int id, string name, double att, double mid, double fin) {
        Id = id; Name = name; Attendance = att; Midterm = mid; Final = fin;
    }
    public double GetTotal() => Attendance + Midterm + Final;
    public string GetGrade() {
        double t = GetTotal();
        if (t >= 85) return "A"; else if (t >= 75) return "B";
        else if (t >= 65) return "C"; else if (t >= 50) return "D";
        else return "F";
    }
}
