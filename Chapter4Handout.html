៤.១ យន្តការ Routing និងការកំណត់ Endpoint ក្នុង Web API
នៅក្នុង ASP.NET Core Web API, Attribute Routing គឺជាវិធីសាស្ត្រស្តង់ដារដែលត្រូវបានណែនាំឱ្យប្រើប្រាស់ជាចម្បង។ វាអនុញ្ញាតឱ្យយើងកំណត់អាសយដ្ឋាន URL ផ្ទាល់នៅលើក្បាល Class របស់ Controller តាមរយៈ Attribute [Route("api/[controller]")]។

• លក្ខណៈយឺតយោង URL៖ និមិត្តសញ្ញា [controller] នឹងដកស្រង់ឈ្មោះរបស់ Class Controller មកជំនួសដោយស្វ័យប្រវត្ត (ឧទាហរណ៍៖ Class StudentController នឹងក្លាយទៅជា URL api/student ដោយស្វ័យប្រវត្តព្រោះប្រព័ន្ធកាត់ពាក្យ Controller ចេញ)។

• ឯករាជ្យភាពនៃ Action URLs៖ យើងអាចបន្ថែមលក្ខខណ្ឌផ្លូវនៅលើក្បាល Action Methods នីមួយៗបន្ថែមទៀត ដូចជា [HttpGet("all")] ឬ [HttpGet("{id}")] ដើម្បីបង្កើត Endpoint ជាក់លាក់។

Attribute Routing Structure
[Route("api/[controller]")]
Prefix ថ្នាក់គ្រឹះ: api/student
[HttpGet("{id}")]
Endpoint កម្រិត Action: api/student/1

៤.៣ ការចាប់យកទិន្នន័យតាមរយៈ Route Parameters
Route Parameters៖ ជាការបង្កប់តម្លៃអថេរ ឬលេខសម្គាល់ (ID) ទៅក្នុងគន្លងផ្លូវ URL ផ្ទាល់តែម្តង។ វាធ្វើឱ្យ URL មានសោភ័ណភាពស្រស់ស្អាត ងាយស្រួលយល់ (RESTful style) និងពេញនិយមបំផុតសម្រាប់កំណត់អត្តសញ្ញាណធនធានទិន្នន័យជាក់លាក់។

• Binding Mechanism: .NET Core ប្រើប្រាស់យន្តការ [FromRoute] ដើម្បីទាញយកតម្លៃអថេរដែលបានបញ្ជាក់នៅក្នុង Route Template (ដូចជា {id}) មកបញ្ចូលទៅក្នុង Parameter របស់ Action Method ដោយស្វ័យប្រវត្ត។

• ឧទាហរណ៍៖ សំណើ URL api/student/3 នឹងផ្គូផ្គងតម្លៃ 3 ទៅឱ្យអថេរ int id នៅក្នុង Action GetById(int id)។

Parameters Binding Sources

[FromQuery]: /api/student?id=1 (ទាញពី Query)

[HttpGet("{id}")] // ផ្គូផ្គងជាមួយ URL: api/student/5
public IActionResult GetById(int id)
{
    var student = _students.Find(s => s.Id == id);
    if (student == null) 
        return NotFound($"រកមិនឃើញនិស្សិតដែលមាន ID = {id} ឡើយ។");
    return Ok(student);
}

៤.៤ ការប្រើប្រាស់ Query String Parameters សម្រាប់តម្រងទិន្នន័យ
Query String Parameters៖ គឺជាប៉ារ៉ាម៉ែត្របន្ថែមនៅខាងចុង URL បន្ទាប់ពីសញ្ញាសួរ (?)។ ពួកគេត្រូវបានប្រើប្រាស់ជាទូទៅសម្រាប់ការងារតម្រងទិន្នន័យ (Filtering) ការស្វែងរក (Searching) ឬការធ្វើទំព័រទិន្នន័យ (Pagination)។
មិនដូច Route Parameters ឡើយ Query String Parameters មិនត្រូវបានប្រកាសនៅក្នុង Route Attribute template ឡើយ។ ប្រព័ន្ធ .NET Core នឹងផ្គូផ្គងឈ្មោះ Parameter របស់ Action Method ទៅនឹង Key ដែលបញ្ជូនមកក្នុង URL Query String ដោយស្វ័យប្រវត្ត (យន្តការ Model Binding)។
កូដគំរូនៃការប្រើប្រាស់ Query String សម្រាប់ស្វែងរកទិន្នន័យ

// URL គំរូ៖ api/student/search?major=Computer Science
[HttpGet("search")]
public IActionResult SearchStudents(string major)
{
    var result = _students.FindAll(s => s.Major.Equals(major, StringComparison.OrdinalIgnoreCase));
    
    if (result.Count == 0)
        return NotFound($"រកមិនឃើញនិស្សិតក្នុងជំនាញ {major} ឡើយ។");

    return Ok(result);
}

៤.៥ ការចាប់យកទិន្នន័យ JSON ពី Request Body ជាមួយ [FromBody]
នៅពេល Client បញ្ជូនទិន្នន័យស្មុគស្មាញ (ដូចជា Object របស់សិស្សដែលមាន Properties ច្រើន) គេច្រើនបញ្ជូនវាមកក្នុង Request Body ជាទម្រង់អត្ថបទ JSON។

[HttpPost] // URL សំណើ៖ POST api/student
public IActionResult CreateStudent([FromBody] Student newStudent)
{
    if (newStudent == null)
        return BadRequest("ទិន្នន័យនិស្សិតមិនអាចទទេឡើយ។");

    _students.Add(newStudent);

    // ត្រឡប់ Status Code 201 Created ត្រឹមត្រូវតាម RESTful Best Practice
    return CreatedAtAction(nameof(GetById), new { id = newStudent.Id }, newStudent);
}

        }

        // ៣. POST: api/student (បញ្ចូលសិស្សថ្មី)

        public IActionResult Update(int id, [FromBody] Student updatedStudent)
        {
            var student = _students.Find(s => s.Id == id);

            student.Major = updatedStudent.Major;

        }

ពន្យល់ពាក្យគន្លឹះ (Glossary)
ពាក្យបច្ចេកទេស និងការពន្យល់ន័យជាភាសាខ្មែរសម្រាប់ជំពូកទី ៤ នេះ៖
ពាក្យបច្ចេកទេស ការពន្យល់ជាភាសាខ្មែរ
Attribute Routing ការកំណត់អាសយដ្ឋាន URL ផ្ទាល់នៅលើក្បាល Class ឬ Method ដោយប្រើប្រាស់ Attribute [Route]។
HTTP Verbs សញ្ញាបញ្ជាក់ពីប្រភេទប្រតិបត្តិការលើគម្រោង Web ដូចជាការទាញទិន្នន័យ (GET) ឬការបញ្ចូល (POST)។
Route Parameters តម្លៃអថេរ ឬទិន្នន័យដែលត្រូវបានបង្កប់ផ្ទាល់នៅក្នុងគន្លងខ្សែស្រឡាយនៃអាសយដ្ឋាន URL។
Query String ប៉ារ៉ាម៉ែត្របន្ថែមនៅខាងចុង URL បន្ទាប់ពីសញ្ញាសួរ (?) ពេញនិយមប្រើសម្រាប់ តម្រង (Filter) ឬស្វែងរក។
Deserialization ដំណើរការបំប្លែងទិន្នន័យអត្ថបទជាទម្រង់ JSON ឱ្យមកជា Objects របស់ភាសា C# វិញ ដើម្បីយកទៅប្រើប្រាស់។
[FromBody] Attribute បញ្ជាក់ថាទិន្នន័យប៉ារ៉ាម៉ែត្រត្រូវទាញចេញពីផ្នែក HTTP Request Body (ជាទូទៅជាទម្រង់ JSON)។
Mock Data ទិន្នន័យសិប្បនិម្មិតដែលបង្កើតឡើងបណ្តោះអាសន្នក្នុងកូដ ដើម្បីតេស្តដំណើរការប្រព័ន្ធ (មិនទាន់មាន Database ពិត)។
201 Created HTTP Status Code ជោគជ័យ បញ្ជាក់ថាធនធានទិន្នន័យថ្មីមួយត្រូវបានបង្កើតឡើងនៅលើ Server ដោយជោគជ័យ។

១. សំណួរ ខុស/ត្រូវ (True/False)

សេចក្តីណែនាំ៖ សូមគូសសញ្ញា (ត្រូវ) ប្រសិនបើប្រយោគខាងក្រោមត្រឹមត្រូវ ឬសញ្ញា (ខុស) ប្រសិនបើខុស។

  1. នៅក្នុង Attribute Routing និមិត្តសញ្ញា [controller] នឹងត្រូវជំនួសដោយឈ្មោះ Class Controller ដោយស្វ័យប្រវត្ត។
  2. Query String Parameters ត្រូវតែប្រកាសនៅក្នុង Attribute Route ជានិច្ច ដូចជា [HttpGet("search/{major}")]។
  3. Attribute [FromBody] ប្រើសម្រាប់ទាញយកទិន្នន័យពី JSON Object ដែលបញ្ជូនមកក្នុង Request Body។
  4. នៅពេលបង្កើតទិន្នន័យថ្មីបានជោគជ័យ ម៉ាស៊ីនមេគួរតែបញ្ជូន Status Code 201 Created ត្រឡប់ទៅវិញ។
  5. HTTP PUT Method ត្រូវបានប្រើប្រាស់ចម្បងសម្រាប់ប្រតិបត្តិការលុបទិន្នន័យ (Delete)។
  6. ASP.NET Core ប្រើប្រាស់បណ្ណាល័យ System.Text.Json ដើម្បីបំប្លែង Objects ទៅជា JSON ដោយស្វ័យប្រវត្ត។
  7. Attribute [FromRoute] ត្រូវបានប្រើសម្រាប់ចាប់យកទិន្នន័យពី Query String (ឧទាហរណ៍៖ ?id=1)។
  8. ការហៅប្រើប្រាស់ CreatedAtAction() នឹងបញ្ជូនមកវិញនូវ Status Code 201 Created និង Location Header នៃ API។
  9. Status Code 204 No Content បញ្ជាក់ថាសំណើបានជោគជ័យ ប៉ុន្តែគ្មានទិន្នន័យណាមួយត្រូវបញ្ជូនត្រឡប់ទៅ Client វិញឡើយ។

// ៥.១ ស្វែងរកតាមជំនាញសិក្សា (Query String)
[HttpGet("search")]
public IActionResult SearchByMajor([FromQuery] string major)
{
    var result = _students.FindAll(s => s.Major.Equals(major, StringComparison.OrdinalIgnoreCase));
    return Ok(result);
}

// ៥.៣ កំណត់លក្ខខណ្ឌ ID ជាចំនួនគត់ (Route Constraints)
[HttpGet("{id:int}")]
public IActionResult GetByIdWithConstraint(int id)
{
    var student = _students.Find(s => s.Id == id);
    if (student == null) return NotFound();
    return Ok(student);
}
លំហាត់ 5.4 (Add Multiple) & លំហាត់ 5.5 (Partial Update)
Controllers/StudentController.cs (Solutions Continued)
// ៥.៤ បញ្ចូលសិស្សច្រើននាក់ក្នុងពេលតែមួយ (POST Body)
[HttpPost("multiple")]
public IActionResult CreateMultiple([FromBody] List<Student> students)
{
    if (students == null || students.Count == 0) 
        return BadRequest("ទិន្នន័យមិនអាចទទេឡើយ");

    _students.AddRange(students);
    return Ok($"បានបញ្ចូលសិស្សចំនួន {students.Count} នាក់ដោយជោគជ័យ");
}

// ៥.៥ ធ្វើបច្ចុប្បន្នភាពតែជំនាញសិក្សា (Partial Update)
[HttpPut("{id}/major")]
public IActionResult UpdateMajor(int id, [FromBody] string newMajor)
{
    var student = _students.Find(s => s.Id == id);
    if (student == null) return NotFound("រកមិនឃើញសិស្សឡើយ");

    student.Major = newMajor;
    return Ok(student);
}
