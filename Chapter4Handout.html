  • бһҷбһӣбҹӢбһ…бҹ’бһ”бһ¶бһҹбҹӢбһ–бһё HTTP Verbs: бһ”бҹӮбһ„бһ…бҹӮбһҖбһҠбһ¶бһ…бҹӢбһҹбҹ’бһҡбһЎбҹҮбһ–бһёбһҸбһҪбһ“бһ¶бһ‘бһёбһҡбһ”бһҹбҹӢ GET, POST, PUT, DELETE бһҹбҹ’бһҡбһ”бһҸбһ¶бһҳбһ”бҹ’бһҡбһҸбһ·бһ”бһҸбҹ’бһҸбһ·бһҖбһ¶бһҡ CRUDбҹ”
  • бһ…бһ¶бһ”бҹӢбһҷбһҖ Parameters бһҹбҹ’бһ‘бһ¶бһҸбҹӢбһҮбҹҶбһ“бһ¶бһү: бһ…бҹҒбҹҮбһ‘бһ¶бһүбһҷбһҖбһўбһҗбҹҒбһҡбһ–бһё Route Path, Query String бһ“бһ·бһ„ Request Body (JSON Object) бһҠбһҫбһҳбҹ’бһ”бһёбһҷбһҖбһ‘бҹ…бһҠбҹҶбһҺбһҫбһҡбһҖбһ¶бһҡ Logic бһҖбҹ’бһ“бһ»бһ„ C#бҹ”

  1. бһ“бҹ…бһҖбҹ’бһ“бһ»бһ„ Attribute Routing бһ“бһ·бһҳбһ·бһҸбҹ’бһҸбһҹбһүбҹ’бһүбһ¶ [controller] бһ“бһ№бһ„бһҸбҹ’бһҡбһјбһңбһҮбҹҶбһ“бһҪбһҹбһҠбҹ„бһҷбһҲбҹ’бһҳбҹ„бҹҮ Class Controller бһҠбҹ„бһҷбһҹбҹ’бһңбҹҗбһҷбһ”бҹ’бһҡбһңбһҸбҹ’бһҸбҹ”
  2. Query String Parameters бһҸбҹ’бһҡбһјбһңбһҸбҹӮбһ”бҹ’бһҡбһҖбһ¶бһҹбһ“бҹ…бһҖбҹ’бһ“бһ»бһ„ Attribute Route бһҮбһ¶бһ“бһ·бһ…бҹ’бһ… бһҠбһјбһ…бһҮбһ¶ [HttpGet("search/{major}")]бҹ”
  3. Attribute [FromBody] бһ”бҹ’бһҡбһҫбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһ‘бһ¶бһүбһҷбһҖбһ‘бһ·бһ“бҹ’бһ“бһ“бҹҗбһҷбһ–бһё JSON Object бһҠбҹӮбһӣбһ”бһүбҹ’бһҮбһјбһ“бһҳбһҖбһҖбҹ’бһ“бһ»бһ„ Request Bodyбҹ”
  4. бһ“бҹ…бһ–бҹҒбһӣбһ”бһ„бҹ’бһҖбһҫбһҸбһ‘бһ·бһ“бҹ’бһ“бһ“бҹҗбһҷбһҗбҹ’бһҳбһёбһ”бһ¶бһ“бһҮбҹ„бһӮбһҮбҹҗбһҷ бһҳбҹүбһ¶бһҹбҹҠбһёбһ“бһҳбҹҒбһӮбһҪбһҡбһҸбҹӮбһ”бһүбҹ’бһҮбһјбһ“ Status Code 201 Created бһҸбҹ’бһҡбһЎбһ”бҹӢбһ‘бҹ…бһңбһ·бһүбҹ”
  5. HTTP PUT Method бһҸбҹ’бһҡбһјбһңбһ”бһ¶бһ“бһ”бҹ’бһҡбһҫбһ”бҹ’бһҡбһ¶бһҹбҹӢбһ…бһҳбҹ’бһ”бһ„бһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһ”бҹ’бһҡбһҸбһ·бһ”бһҸбҹ’бһҸбһ·бһҖбһ¶бһҡбһӣбһ»бһ”бһ‘бһ·бһ“бҹ’бһ“бһ“бҹҗбһҷ (Delete)бҹ”
  6. ASP.NET Core бһ”бҹ’бһҡбһҫбһ”бҹ’бһҡбһ¶бһҹбҹӢбһ”бһҺбҹ’бһҺбһ¶бһӣбҹҗбһҷ System.Text.Json бһҠбһҫбһҳбҹ’бһ”бһёбһ”бҹҶбһ”бҹ’бһӣбҹӮбһ„ Objects бһ‘бҹ…бһҮбһ¶ JSON бһҠбҹ„бһҷбһҹбҹ’бһңбҹҗбһҷбһ”бҹ’бһҡбһңбһҸбҹ’бһҸбҹ”
  7. Attribute [FromRoute] бһҸбҹ’бһҡбһјбһңбһ”бһ¶бһ“бһ”бҹ’бһҡбһҫбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһ…бһ¶бһ”бҹӢбһҷбһҖбһ‘бһ·бһ“бҹ’бһ“бһ“бҹҗбһҷбһ–бһё Query String (бһ§бһ‘бһ¶бһ бһҡбһҺбҹҚбҹ– ?id=1)бҹ”
  8. бһҖбһ¶бһҡбһ бҹ…бһ”бҹ’бһҡбһҫбһ”бҹ’бһҡбһ¶бһҹбҹӢ CreatedAtAction() бһ“бһ№бһ„бһ”бһүбҹ’бһҮбһјбһ“бһҳбһҖбһңбһ·бһүбһ“бһјбһң Status Code 201 Created бһ“бһ·бһ„ Location Header бһ“бҹғ APIбҹ”
  9. Status Code 204 No Content бһ”бһүбҹ’бһҮбһ¶бһҖбҹӢбһҗбһ¶бһҹбҹҶбһҺбһҫбһ”бһ¶бһ“бһҮбҹ„бһӮбһҮбҹҗбһҷ бһ”бҹүбһ»бһ“бҹ’бһҸбҹӮбһӮбҹ’бһҳбһ¶бһ“бһ‘бһ·бһ“бҹ’бһ“бһ“бҹҗбһҷбһҺбһ¶бһҳбһҪбһҷбһҸбҹ’бһҡбһјбһңбһ”бһүбҹ’бһҮбһјбһ“бһҸбҹ’бһҡбһЎбһ”бҹӢбһ‘бҹ… Client бһңбһ·бһүбһЎбһҫбһҷбҹ”
  10. бһҷбһҫбһ„бһҳбһ·бһ“бһўбһ¶бһ…бһҖбҹҶбһҺбһҸбҹӢбһӣбһҖбҹ’бһҒбһҒбһҺбҹ’бһҢбһ”бҹ’бһҡбһ—бҹҒбһ‘бһўбһҗбҹҒбһҡ (Type Constraints) бһ“бҹ…бһӣбһҫ Route Parameter бһ”бһ¶бһ“бһЎбһҫбһҷ (бһҠбһјбһ…бһҮбһ¶бҹ– {id:int})бҹ”

// бҹҘ.бҹЎ бһҹбҹ’бһңбҹӮбһ„бһҡбһҖбһҸбһ¶бһҳбһҮбҹҶбһ“бһ¶бһүбһҹбһ·бһҖбҹ’бһҹбһ¶ (Query String)
[HttpGet("search")]
public IActionResult SearchByMajor([FromQuery] string major)
{
    var result = _students.FindAll(s => s.Major.Equals(major, StringComparison.OrdinalIgnoreCase));
    return Ok(result);
}

// бҹҘ.бҹЈ бһҖбҹҶбһҺбһҸбҹӢбһӣбһҖбҹ’бһҒбһҒбһҺбҹ’бһҢ ID бһҮбһ¶бһ…бҹҶбһ“бһҪбһ“бһӮбһҸбҹӢ (Route Constraints)
[HttpGet("{id:int}")]
public IActionResult GetByIdWithConstraint(int id)
{
    var student = _students.Find(s => s.Id == id);
    if (student == null) return NotFound();
    return Ok(student);
}

// бҹҘ.бҹӨ бһ”бһүбҹ’бһ…бһјбһӣбһҹбһ·бһҹбҹ’бһҹбһ…бҹ’бһҡбһҫбһ“бһ“бһ¶бһҖбҹӢбһҖбҹ’бһ“бһ»бһ„бһ–бҹҒбһӣбһҸбҹӮбһҳбһҪбһҷ (POST Body)
[HttpPost("multiple")]
public IActionResult CreateMultiple([FromBody] List<Student> students)
{
    if (students == null || students.Count == 0) 
        return BadRequest("бһ‘бһ·бһ“бҹ’бһ“бһ“бҹҗбһҷбһҳбһ·бһ“бһўбһ¶бһ…бһ‘бһ‘бҹҒбһЎбһҫбһҷ");

    _students.AddRange(students);
    return Ok($"бһ”бһ¶бһ“бһ”бһүбҹ’бһ…бһјбһӣбһҹбһ·бһҹбҹ’бһҹбһ…бҹҶбһ“бһҪбһ“ {students.Count} бһ“бһ¶бһҖбҹӢбһҠбҹ„бһҷбһҮбҹ„бһӮбһҮбҹҗбһҷ");
}

// бҹҘ.бҹҘ бһ’бҹ’бһңбһҫбһ”бһ…бҹ’бһ…бһ»бһ”бҹ’бһ”бһ“бҹ’бһ“бһ—бһ¶бһ–бһҸбҹӮбһҮбҹҶбһ“бһ¶бһүбһҹбһ·бһҖбҹ’бһҹбһ¶ (Partial Update)
[HttpPut("{id}/major")]
public IActionResult UpdateMajor(int id, [FromBody] string newMajor)
{
    var student = _students.Find(s => s.Id == id);
    if (student == null) return NotFound("бһҡбһҖбһҳбһ·бһ“бһғбһҫбһүбһҹбһ·бһҹбҹ’бһҹбһЎбһҫбһҷ");

    student.Major = newMajor;
    return Ok(student);
}
