using Microsoft.AspNetCore.Mvc;
using System;

namespace OopMvcApplication.Controllers
{
    [ApiController]
    [Route("api/[controller]")]
    public class AppController : ControllerBase
    {
        // ១. Action ទី១៖ បញ្ជូនឈ្មោះកម្មវិធីត្រឡប់ទៅវិញជា String
        [HttpGet("name")]
        public IActionResult GetAppName()
        {
            return Ok("OopMvcApplication (Web API Version 1.0)");
        }

        // ២. Action ទី២៖ បញ្ជូនពេលវេលារបស់ Server ត្រឡប់ទៅវិញជា JSON
        [HttpGet("servertime")]
        public IActionResult GetServerTime()
        {
            var timeInfo = new 
            { 
                currentTime = DateTime.Now.ToString("yyyy-MM-dd HH:mm:ss"),
                timeZone = TimeZoneInfo.Local.DisplayName
            };
            return Ok(timeInfo);
        }
    }
}

៣.២ Controller Classes & Action Methods (IActionResult)
Action Method គឺជា public method នៅក្នុង Controller ដែលរត់រាល់ពេលមានសំណើ HTTP Request ត្រូវគ្នាមកដល់។ Action Method នីមួយៗត្រូវបញ្ជូនលទ្ធផលត្រឡប់ទៅវិញជាប្រភេទ IActionResult ដែលអនុញ្ញាតឱ្យយើងបញ្ជូនទិន្នន័យរួមជាមួយ Status Code របស់ HTTP (ដូចជា 200 OK, 404 Not Found)។
ប្រភេទ IActionResult និង HTTP Status Codes សំខាន់ៗ
ប្រភេទ Result HTTP Status អត្ថន័យបច្ចេកទេស និងការប្រើប្រាស់ក្នុង API
Ok(data) 200 OK សំណើជោគជ័យ និងបញ្ជូនទិន្នន័យត្រឡប់ទៅ Client។ ប្រើប្រាស់ជាទូទៅបំផុត។
Created(uri, data) 201 Created បង្កើតទិន្នន័យថ្មីក្នុង DB ជោគជ័យ។ ប្រើប្រាស់បន្ទាប់ពីការសរសេរកូដបញ្ចូលទិន្នន័យថ្មី (POST)។
BadRequest() 400 Bad Request ទម្រង់សំណើផ្ញើមកខុស ឬមិនពេញលេញ។ ប្រើប្រាស់នៅពេល Client បញ្ជូនទិន្នន័យមកខុសលក្ខខណ្ឌ។
NotFound() 404 Not Found រកមិនឃើញទិន្នន័យដែលស្នើសុំ។ ប្រើប្រាស់ពេលស្វែងរកទិន្នន័យតាម ID រួចរកមិនឃើញក្នុង DB។

using Microsoft.AspNetCore.Mvc;
using System.Collections.Generic;
using OopMvcApplication.Models; // នាំចូល Class Model 'Student'

namespace OopMvcApplication.Controllers
{
    [ApiController]
    [Route("api/[controller]")]
    public class StudentController : ControllerBase
    {
        // Action Method បញ្ចេញបញ្ជីសិស្សទាំងអស់ជា JSON Array
        [HttpGet]
        public IActionResult GetStudents()
        {
            // ១. បង្កើតបញ្ជីសិស្សសិប្បនិម្មិត (Mock Data List)
            List<Student> students = new()
            {
                new Student(1, "Dara SOK", "Computer Science"),
                new Student(2, "Cheata SENG", "Information Technology"),
                new Student(3, "Sokha PHEN", "Cyber Security")
            };

            // ២. បញ្ជូនបញ្ជីសិស្សទៅកាន់ Client ក្នុងទម្រង់ JSON Array
            return Ok(students);
        }
    }
}

៣.៣ ដំណើរការ Routing & Endpoint Basics (Attribute Routing)
Routing គឺជាយន្តការស្វែងរក និងផ្គូផ្គង URL របស់សំណើដែលផ្ញើមកពី Browser ទៅកាន់ Action Method ជាក់លាក់របស់ Controller នៅក្នុង Server។ នៅក្នុង ASP.NET Core API គេនិយមប្រើប្រាស់ Attribute Routing (ការសរសេរ Route នៅលើក្បាល Class ឬ Method ផ្ទាល់) ព្រោះវាផ្តល់ភាពច្បាស់លាស់ និងងាយស្រួលគ្រប់គ្រង។
ការប្រើប្រាស់ Attribute Routing លើ Controller & Action Methods
// URL Route របស់ Controller ត្រូវបានកំណត់តាម [controller] ដែលយកឈ្មោះខាងមុខ Class មកធ្វើជា Route URL
[Route("api/[controller]")] 
public class ProductController : ControllerBase 
{
    // URL Endpoint: api/product
    [HttpGet]
    public IActionResult GetAllProducts() { ... }

    // URL Endpoint: api/product/promo
    [HttpGet("promo")]
    public IActionResult GetPromoProducts() { ... }
}

Action Method Public method ក្នុង Controller ដែលបង្កើតឡើងដើម្បីទទួលសំណើ និងបញ្ជូនលទ្ធផលទៅកាន់ Client។
IActionResult Interface កំណត់លទ្ធផលការងាររបស់ Action Method អាចបញ្ជូនទិន្នន័យរួមគ្នាជាមួយ HTTP Status Code។

សេចក្តីសង្ខេបជំពូកទី ៣ (Summary)
  • Controllers: គឺជាខួរក្បាល និងជាអ្នកសម្របសម្រួលលំហូរទិន្នន័យដ៏សំខាន់បំផុតរបស់គម្រោង ASP.NET Core MVC។
  • ControllerBase: Web API Controller ត្រូវតែស្នង (Inherit) ពី ControllerBase និងកំណត់ Attribute [ApiController] ជានិច្ច។
  • Action Methods & IActionResult: រាល់ Action ទាំងអស់ត្រូវបញ្ជូនលទ្ធផលត្រឡប់ទៅវិញជាប្រភេទ IActionResult (ដូចជា Ok(), Created(), NotFound())។
  • Attribute Routing: ការកំណត់ URL Route ផ្ទាល់លើ Action ជួយផ្តល់ភាពងាយស្រួល និងរៀបចំ endpoints បានច្បាស់លាស់។
  • JSON Serialization: ASP.NET Core ប្រើប្រាស់ System.Text.Json ដើម្បីបំប្លែង C# Objects ទៅជា JSON Response ដោយស្វ័យប្រវត្ត។

១. សំណួរ ខុស/ត្រូវ (True/False)

សេចក្តីណែនាំ៖ សូមគូសសញ្ញា (ត្រូវ) ប្រសិនបើប្រយោគខាងក្រោមត្រឹមត្រូវ ឬសញ្ញា (ខុស) ប្រសិនបើខុស។

  1. Web API Controllers របស់ ASP.NET Core គួរតែស្នង (Inherit) ពី class Controller ជានិច្ច។
  2. Attribute [ApiController] ជួយបំពេញការងារត្រួតពិនិត្យ validation ទិន្នន័យដោយស្វ័យប្រវត្ត។
  3. Action Method ទាំងអស់នៅក្នុង Web API ត្រូវតែបញ្ជូនត្រឡប់ជាប្រភេទ IActionResult ជានិច្ច។
  4. យន្តការ Ok(data) បញ្ជូន Status Code 400 Bad Request ត្រឡប់ទៅកាន់ Client។
  5. Attribute Routing គឺជាការសរសេរ Route URLs នៅលើក្បាល class ឬ method របស់ Controller ផ្ទាល់។
  6. បណ្ណាល័យ System.Text.Json បំប្លែង C# Objects ទៅជា JSON Response ដោយស្វ័យប្រវត្ត។
  7. សញ្ញា {} នៅក្នុងទម្រង់ JSON ត្រូវបានប្រើប្រាស់សម្រាប់សម្គាល់ Array (បញ្ជីសមាជិក)។
  8. សន្ទស្សន៍ URL endpoint ត្រូវបានកំណត់នៅក្នុង Attribute [Route] ឬ [HttpGet]។
  9. លទ្ធផល Http Status Code 404 សម្គាល់ថា ម៉ាស៊ីនមេត្រូវបានដំឡើងជោគជ័យ (Created)។
  10. Mock Data ត្រូវបានបង្កើតឡើងនៅក្នុង Controller ដើម្បីតេស្តប្រព័ន្ធជាបណ្ដោះអាសន្ន។

ក. កំរិតស្រាល (Easy Level) - ២ លំហាត់
  • លំហាត់ 5.1 (WelcomeApi)៖ ចូរសរសេរ API Controller ថ្មីមួយឈ្មោះ WelcomeController ដែលមាន Action Method មួយឈ្មោះ GetMessage() បញ្ជូនសារ "Welcome to ASP.NET Core Web API!" ត្រឡប់ទៅ Client វិញជា String (Status 200 OK)។
  • លំហាត់ 5.2 (PingApi)៖ ចូរសរសេរ API Controller ថ្មីមួយឈ្មោះ PingController ដែលមាន Action Method មួយឈ្មោះ Ping() បញ្ជូន JSON Object ត្រឡប់ទៅ Client វិញដែលមានផ្ទុកព័ត៌មាន status = "Healthy" និង timestamp = [ពេលវេលាបច្ចុប្បន្ន]។
ខ. កំរិតមធ្យម (Medium Level) - ២ លំហាត់
  • លំហាត់ 5.3 (ColorApi)៖ ចូរសរសេរ Action Method មួយឈ្មោះ GetColors() នៅក្នុង ColorController ដែលប្រកាសបញ្ជីពណ៌ (List of strings) រួមមាន "Red", "Blue", "Green", "Yellow" រួចបញ្ជូនទិន្នន័យបញ្ជីពណ៌ទាំងនោះជាទម្រង់ JSON Array (Status 200 OK)។
  • លំហាត់ 5.4 (DoctorApi)៖ ចូរបង្កើត Class Model មួយឈ្មោះ Doctor មាន Properties: Id, Name, Specialty និង Constructor ផ្ដល់តម្លៃដំបូង។ បន្ទាប់មកបង្កើត DoctorController ដែលមាន Action Method មួយ បង្កើត Object របស់ Doctor នេះ រួចបញ្ជូនជាទម្រង់ JSON ត្រឡប់ទៅ Client។
គ. កំរិតធ្ងន់ (Hard Level) - ២ លំហាត់
  • លំហាត់ 5.5 (DepartmentApi)៖ ចូរបង្កើត Class Model មួយឈ្មោះ Department មាន Properties: Id, Name, និង Employees (List of strings)។ នៅក្នុង DepartmentController ត្រូវបង្កើត Action Method មួយ បង្កើត Object របស់ Department នេះ (ដែលមានបញ្ចូលឈ្មោះបុគ្គលិក ៣ នាក់) រួចបញ្ជូនជាទម្រង់ JSON Object ដែលមានផ្ទុកបញ្ជីឈ្មោះបុគ្គលិកនៅផ្នែកខាងក្នុង។
  • លំហាត់ 5.6 (BookStoreApi)៖ ចូរបង្កើត Class Model មួយឈ្មោះ Book មាន Properties: Id, Title, Author, Price។ នៅក្នុង Book Class ត្រូវបង្កើត Method មួយឈ្មោះ GetDiscountedPrice() ដែលគណនាបញ្ចុះតម្លៃ ១០% លើតម្លៃសៀវភៅ។ សរសេរ BookController បង្កើតបញ្ជីសៀវភៅ ៣ ក្បាល (Mock Data List) រួចបញ្ជូនទៅ Client ជាទម្រង់ JSON Array ដែលមានផ្ទុកតម្លៃ Price ដើម និង DiscountedPrice គណនាភ្លាមៗ។
