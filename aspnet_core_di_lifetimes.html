ការយល់ដឹងស៊ីជម្រៅអំពី ASP.NET Core Service Lifetimes

មគ្គុទ្ទេសក៍លម្អិត អមដោយឧទាហរណ៍កូដជាក់ស្តែងសម្រាប់ស្ថាបត្យកម្មកម្រិត Enterprise

💡 អំពី Dependency Injection (DI): នៅក្នុង ASP.NET Core ការគ្រប់គ្រងអាយុកាលរបស់ Object (Instance Lifetimes) គឺមានសារៈសំខាន់បំផុត ដើម្បីធានាបាននូវល្បឿន លំនឹងប្រព័ន្ធ និងការប្រើប្រាស់ Memory ប្រកបដោយប្រសិទ្ធភាពខ្ពស់។ ខាងក្រោមនេះជាការវិភាគលម្អិត និងការអនុវត្តកូដជាក់ស្តែង។

១. Transient (អាយុកាលខ្លីបំផុត / បង្កើតថ្មីជានិច្ច)

Transient

🔴 ទ្រឹស្តី និងជីវិតពិត

ទ្រឹស្តី៖ ប្រព័ន្ធ DI container នឹងបង្កើត Object ថ្មីស្រឡាងរាល់ពេលដែលមានការហៅប្រើប្រាស់ (Injection) ទោះបីជាស្ថិតនៅក្នុង Request តែមួយ ឬ Request ផ្សេងគ្នាក៏ដោយ។

ឧទាហរណ៍៖ កែវក្រដាសប្រើម្តងបោះចោល។ គ្រប់ពេលដែលចង់ញ៉ាំទឹក គឺត្រូវដកកែវថ្មីមកប្រើជានិច្ច មិនប្រើរួមគ្នាឡើយ។

🛠️ ករណីប្រើប្រាស់សមស្រប

ប្រើសម្រាប់សេវាកម្មដែលជាប្រភេទ Stateless (មិនរក្សាទុកស្ថានភាពទិន្នន័យ) ស្រាលៗ និងដំណើរការលឿន ដូចជា៖ សេវាកម្មជំនួយការគណនា, ជំនួយការបង្កើត UUID/Random String, ឬ Validation Services។

💻 ឧទាហរណ៍កូដការអនុវត្ត (Implementation):
public interface ITransientService { Guid GetOperationId(); }

public class TransientService : ITransientService
{
    private readonly Guid _id = Guid.NewGuid();
    public Guid GetOperationId() => _id;
}

២. Scoped (អាយុកាលកម្រិត Request / តែមួយគត់ក្នុងមួយសំណើ)

Scoped

🟢 ទ្រឹស្តី និងជីវិតពិត

ទ្រឹស្តី៖ Object ត្រូវបានបង្កើតឡើងតែម្តងគត់ក្នុងមួយ HTTP Request។ គ្រប់សមាសធាតុ (Classes/Services) ទាំងអស់ដែលហៅប្រើប្រាស់វានៅក្នុង Request តែមួយនោះ នឹងចែករំលែក Instance តែមួយរួមគ្នា។

ឧទាហរណ៍៖ អ្នករត់តុបម្រើអតិថិជនតាមតុ។ ក្នុងអំឡុងពេលដែលអ្នកអង្គុយពិសារអាហារ (១ Request) គឺមានអ្នករត់តុដដែលបម្រើអ្នកតាំងពីដើមរហូតដល់ចប់។

🛠️ ករណីប្រើប្រាស់សមស្រប

ប្រើសម្រាប់សេវាកម្មដែលត្រូវរក្សាទុក State ក្នុងរង្វង់ Request មួយ ឬសេវាកម្មដែលទាក់ទងនឹង Database Connection ដូចជា DbContext របស់ Entity Framework Core, Repositories, ឬ Business Services (Unit of Work)។

💻 ឧទាហរណ៍កូដការអនុវត្ត (Implementation):
public interface IScopedService { Guid GetOperationId(); }

public class ScopedService : IScopedService
{
    private readonly Guid _id = Guid.NewGuid();
    public Guid GetOperationId() => _id;
}

៣. Singleton (អាយុកាលវែងបំផុត / រួមគ្នាពេញមួយកម្មវិធី)

Singleton

🔵 ទ្រឹស្តី និងជីវិតពិត

ទ្រឹស្តី៖ Object ត្រូវបានបង្កើតឡើងតែម្តងគត់នៅពេលដែលកម្មវិធីចាប់ផ្តើម (Application Startup) ឬនៅពេលត្រូវបានហៅប្រើប្រាស់លើកដំបូង ហើយ Instance នោះនឹងត្រូវប្រើប្រាស់រួមគ្នាសម្រាប់គ្រប់ Request និងគ្រប់ User ទាំងអស់រហូតដល់ App ត្រូវបានបិទ។

ឧទាហរណ៍៖ នាឡិកាព្យួរជញ្ជាំងក្នុងក្រុមហ៊ុន។ គ្រប់បុគ្គលិកទាំងអស់ (រាល់ Request ទាំងអស់) មើលម៉ោងពីនាឡិកាតែមួយរួមគ្នា។

🛠️ ករណីប្រើប្រាស់សមស្រប

ប្រើសម្រាប់សេវាកម្មដែលចំណាយធនធានច្រើនក្នុងការបង្កើត (Expensive to create) ឬត្រូវការរក្សាទុកទិន្នន័យជាសកល (Global State Cache) ដូចជា៖ Memory Cache, Configuration Settings, ឬ Message Broker Client (RabbitMQ Connection)។

💻 ឧទាហរណ៍កូដការអនុវត្ត (Implementation):
public interface ISingletonService { Guid GetOperationId(); }

public class SingletonService : ISingletonService
{
    private readonly Guid _id = Guid.NewGuid();
    public Guid GetOperationId() => _id;
}

⚙️ ការចុះឈ្មោះសេវាកម្ម និងការប្រើប្រាស់ក្នុង Web API

ដើម្បីមើលឃើញភាពខុសគ្នាជាក់ស្តែង យើងបង្កើត Web API Controller មួយដែលហៅសេវាកម្មទាំងនេះចំនួន ២ ដងក្នុង Request តែមួយ៖

// ១. ការចុះឈ្មោះក្នុង Program.cs
var builder = WebApplication.CreateBuilder(args);
builder.Services.AddTransient<ITransientService, TransientService>();
builder.Services.AddScoped<IScopedService, ScopedService>();
builder.Services.AddSingleton<ISingletonService, SingletonService>();

// ២. បង្កើត Web API Controller ដើម្បីធ្វើតេស្ត
[ApiController]
[Route("api/[controller]")]
public class LifetimesController : ControllerBase
{
    private readonly ITransientService _transient1;
    private readonly ITransientService _transient2;
    private readonly IScopedService _scoped1;
    private readonly IScopedService _scoped2;
    private readonly ISingletonService _singleton1;
    private readonly ISingletonService _singleton2;

    public LifetimesController(
        ITransientService transient1, ITransientService transient2,
        IScopedService scoped1, IScopedService scoped2,
        ISingletonService singleton1, ISingletonService singleton2)
    {
        _transient1 = transient1; _transient2 = transient2;
        _scoped1 = scoped1; _scoped2 = scoped2;
        _singleton1 = singleton1; _singleton2 = singleton2;
    }

    [HttpGet]
    public IActionResult Get()
    {
        return Ok(new
        {
            Transient_1 = _transient1.GetOperationId(),
            Transient_2 = _transient2.GetOperationId(), // នឹងបង្ហាញលទ្ធផលខុសគ្នាដាច់
            Scoped_1 = _scoped1.GetOperationId(),
            Scoped_2 = _scoped2.GetOperationId(), // នឹងបង្ហាញលទ្ធផលដូចគ្នា
            Singleton_1 = _singleton1.GetOperationId(),
            Singleton_2 = _singleton2.GetOperationId() // នឹងបង្ហាញលទ្ធផលដូចគ្នា ទោះហៅពី Request ផ្សេងក៏ដោយ
        });
    }
}

📊 តារាងប្រៀបធៀបសង្ខេប (Comparison Matrix)

ប្រភេទអាយុកាល (Lifetime) ចំនួន Instance ក្នុងមួយ Request ចំនួន Instance ច្រើន Request ការប្រើប្រាស់ Memory
Transient បង្កើតថ្មីរាល់ពេលហៅ (ច្រើន) បង្កើតថ្មីជានិច្ច ទាប (ដោះលែងមកវិញភ្លាមៗក្រោយប្រើ)
Scoped តែមួយគត់ (Instance តែមួយ) បង្កើតថ្មីសម្រាប់ Request ថ្មី មធ្យម (ដោះលែងក្រោយចប់ Request)
Singleton តែមួយគត់ (Instance តែមួយ) រក្សាទុក Instance ដដែលដដែល ខ្ពស់/ថេរ (ស្ថិតក្នុង Memory រហូត)

អភិវឌ្ឍន៍ និងរៀបចំរចនាសម្ព័ន្ធដោយយកចិត្តទុកដាក់ខ្ពស់សម្រាប់ប្រព័ន្ធកម្មវិធីល្បឿនលឿន។
