អ្វីទៅជា ASP.NET Core?
ASP.NET Core គឺជា Framework រងរបស់ .NET ដែលត្រូវបានរចនាឡើងយ៉ាងពិសេសសម្រាប់តែការបង្កើត Web Applications និង Application Programming Interface (API) ប៉ុណ្ណោះ។ វារួមបញ្ចូលនូវឧបករណ៍ស្រាប់ៗ (Built-in Tools) ដូចជា ការគ្រប់គ្រងប្រព័ន្ធសុវត្ថិភាព (Authentication), ការកំណត់ផ្លូវ (Routing), និងការភ្ជាប់ជាមួយប្រព័ន្ធមូលដ្ឋានទិន្នន័យ SQL Server បានយ៉ាងងាយស្រួល។
១.២ ការស្វែងយល់ពី Application Programming Interface (API)

១.២ ករណីសិក្សា៖ អ្នករត់តុក្នុងហាងអាហារ (API Analogy)
ការប្រៀបធៀបដើម្បីងាយស្រមៃ៖
១. ភ្ញៀវ (Client): អ្នកប្រើប្រាស់ Web ឬ Mobile App ដែលត្រូវការទិន្នន័យ (ម្ហូប)។
២. អ្នករត់តុ (API): ទទួលសំណើ (Request/ការកម្ម៉ង់) ពីភ្ញៀវ រួចយកបញ្ជីម្ហូបទៅប្រាប់ចុងភៅនៅខាងក្រោយ។
៣. ចុងភៅ (Server/Database): ចាត់ចែងរៀបចំទិន្នន័យដែលស្នើសុំ (ធ្វើម្ហូប) រួចផ្ញើអាហារ (Data/JSON Response) ត្រឡប់មកវិញតាមរយៈអ្នករត់តុ (API)។
ប្រភេទនៃ API Methods (RESTful HTTP Verbs)
នៅក្នុងការរចនាប្រព័ន្ធ RESTful API យើងប្រើប្រាស់ HTTP Verbs ស្តង់ដារដើម្បីកំណត់ប្រភេទប្រតិបត្តិការនីមួយៗលើទិន្នន័យ៖
HTTP Verb ប្រតិបត្តិការ ការពន្យល់លម្អិត
GET អាន (Read) ប្រើសម្រាប់ស្នើសុំទាញយកទិន្នន័យពី Server មកបង្ហាញ។ ប្រៀបដូចជាការអានសៀវភៅ ឬមើលបញ្ជីឈ្មោះសិស្ស។
POST បញ្ចូល (Create) ប្រើសម្រាប់ផ្ញើទិន្នន័យថ្មីទៅរក្សាទុកក្នុង Database។ ប្រៀបដូចជាការចុះឈ្មោះបង្កើតគណនីថ្មី។
PUT កែប្រែ (Update) ប្រើសម្រាប់ធ្វើបច្ចុប្បន្នភាពព័ត៌មានចាស់ដែលមានស្រាប់ឱ្យទៅជាព័ត៌មានថ្មីទាំងស្រុង។
DELETE លុប (Delete) ប្រើសម្រាប់បញ្ជាលុបទិន្នន័យណាមួយចេញពីប្រព័ន្ធទាំងស្រុង។

រាល់សំណើដែលបញ្ជូនទៅកាន់ Server នឹងទទួលបានលេខកូដឆ្លើយតបមកវិញ ដើម្បីប្រាប់អំពីលទ្ធផលនៃសំណើនោះ៖
Status Code អត្ថន័យ និងការឆ្លើយតបរបស់ប្រព័ន្ធ
200 OK សំណើទទួលបានជោគជ័យ។ ទិន្នន័យត្រូវបានបញ្ជូនត្រឡប់ទៅវិញដោយរលូន។
201 Created ទិន្នន័យថ្មី (POST) ត្រូវបានបង្កើត និងរក្សាទុកក្នុង Database ជោគជ័យ។
400 Bad Request សំណើមានកំហុស។ អ្នកប្រើប្រាស់អាចបញ្ជូនទិន្នន័យខុសទម្រង់ ឬខ្វះចន្លោះ។
404 Not Found រកមិនឃើញទិន្នន័យ ឬអាសយដ្ឋាន API (URL) ដែលបានស្នើសុំនោះទេក្នុងប្រព័ន្ធ។
500 Internal Error មានកំហុសកើតឡើងនៅលើ Server (កូដធ្លាក់ Error) ដែលអ្នកអភិវឌ្ឍន៍ត្រូវជួសជុល។

១.៣ ស្ថាបត្យកម្ម Model-View-Controller (MVC) ពិស្តារ
MVC គឺជាស្ថាបត្យកម្មរចនាសម្ព័ន្ធកូដដែលបែងចែកការងារច្បាស់លាស់ ងាយស្រួលក្នុងការអភិវឌ្ឍ និងថែទាំប្រព័ន្ធធំៗ៖
  • Model (M): តំណាងឱ្យទិន្នន័យ (Data Structure) និងមានតួនាទីទំនាក់ទំនងផ្ទាល់ជាមួយ SQL Server Database។
  • View (V): ផ្ទាំងបង្ហាញ (User Interface) ដែលនៅក្នុង API នេះ View ត្រូវបានជំនួសដោយទម្រង់ទិន្នន័យ JSON Response។
  • Controller (C): ខួរក្បាលអ្នកចាត់ចែង (Logic Controller) ទទួលសំណើ (HTTP Request) ពីអ្នកប្រើប្រាស់ បញ្ជាទៅ Model ឱ្យធ្វើការ រួចទាញទិន្នន័យមកប្រគល់ឱ្យ View។
តើ MVC ធ្វើការជាមួយគ្នាយ៉ាងដូចម្តេច?
១. អ្នកប្រើប្រាស់ធ្វើសកម្មភាព (ឧ: ចុចប៊ូតុងស្វែងរកសិស្ស)។ → ២. សំណើ (HTTP Request) ត្រូវបានបញ្ជូនទៅកាន់ Controller។ → ៣. Controller ពិនិត្យសំណើ រួចបញ្ជាទៅកាន់ Model ឱ្យទៅទាញយកទិន្នន័យក្នុង SQL Database។ → ៤. Model បញ្ជូនទិន្នន័យត្រឡប់មក Controller វិញ។ → ៥. Controller យកទិន្នន័យនោះបំប្លែងជា JSON រួចឆ្លើយតប (Response) ទៅកាន់ View។

២. ឯកសារចាត់ចែង API Logic: Controllers/BooksController.cs
using Microsoft.AspNetCore.Mvc;
using BookApi.Models;

namespace BookApi.Controllers {
    [ApiController]
    [Route("api/[controller]")]
    public class BooksController : ControllerBase {
        private static readonly List<Book> _books = new() {
            new Book { Id = 1, Title = "C# Basics", Author = "Soeurng LIM", Price = 15.50m },
            new Book { Id = 2, Title = "ASP.NET API", Author = "Developer", Price = 25.00m }
        };

        [HttpGet]
        public IActionResult GetAllBooks() => Ok(_books);

        [HttpGet("{id}")]
        public IActionResult GetBookById(int id) {
            var book = _books.FirstOrDefault(b => b.Id == id);
            return book != null ? Ok(book) : NotFound(new { Message = "សៀវភៅរកមិនឃើញ" });
        }
    }
}

លំហាត់នេះតម្រូវឱ្យនិស្សិតបង្កើតគម្រោង ASP.NET Core Web API មួយ ដែលមានមុខងារចុះឈ្មោះសិស្សថ្មីតាមរយៈ HTTP POST Method និងកំណត់ការផ្ទៀងផ្ទាត់លក្ខខណ្ឌទិន្នន័យ (Data Validation)៖
១. ឯកសារ Model ជាមួយ Data Annotations: Models/Student.cs
using System.ComponentModel.DataAnnotations;

namespace StudentApi.Models {
    public class Student {
        public int Id { get; set; }
        [Required(ErrorMessage = "សូមបញ្ចូលឈ្មោះសិស្ស")]
        public string Name { get; set; } = string.Empty;
        [Required(ErrorMessage = "សូមបញ្ចូលជំនាញសិក្សា")]
        public string Major { get; set; } = string.Empty;
        [Range(18, 60, ErrorMessage = "អាយុត្រូវនៅចន្លោះ ១៨ ដល់ ៦០ ឆ្នាំ")]
        public int Age { get; set; }
    }
}
២. ឯកសារ Controller ជាមួយ POST validation: Controllers/StudentsController.cs
using Microsoft.AspNetCore.Mvc;
using StudentApi.Models;

namespace StudentApi.Controllers {
    [ApiController]
    [Route("api/[controller]")]
    public class StudentsController : ControllerBase {
        private static readonly List<Student> _students = new();

        [HttpPost]
        public IActionResult RegisterStudent([FromBody] Student newStudent) {
            if (!ModelState.IsValid) {
                return BadRequest(ModelState); // ត្រឡប់ Error 400 Bad Request
            }
            newStudent.Id = _students.Count > 0 ? _students.Max(s => s.Id) + 1 : 1;
            _students.Add(newStudent);
            return Ok(newStudent);
        }
    }
}
