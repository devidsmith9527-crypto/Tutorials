៥.៤ ការកសាង Class DbContext និងការប្រើប្រាស់ DbSet
នៅក្នុង Entity Framework Core, DbContext គឺជា Class ស្នូលដែលចាត់ចែងរាល់កិច្ចការងារពាក់ព័ន្ធនឹងការតភ្ជាប់ទៅកាន់ Database។ វាតំណាងឱ្យការបើកសម័យកាលធ្វើការ (Session) មួយរវាងកម្មវិធី C# និង Database SQL Server។ រាល់តារាង (Tables) ដែលយើងចង់បង្កើត ត្រូវតែប្រកាសជា Property ប្រភេទ DbSet<TEntity> នៅក្នុង Class នេះ។
រចនាសម្ព័ន្ធកូដ AppDbContext ពេញលេញ៖
Data/AppDbContext.cs
using Microsoft.EntityFrameworkCore;
using SchoolApp.Models;

namespace SchoolApp.Data
{
    // ត្រូវតែស្នងត្រកូលពី class DbContext របស់ EF Core
    public class AppDbContext : DbContext
    {
        // Constructor សម្រាប់ទទួលយកការកំណត់ (Connection String) ពី Program.cs
        public AppDbContext(DbContextOptions<AppDbContext> options) : base(options)
        {
        }

        // ប្រកាស DbSet: Class 'Student' នឹងត្រូវបម្លែងជា Table ឈ្មោះ 'Students'
        public DbSet<Student> Students { get; set; }
    }
}
យន្តការបម្លែងទិន្នន័យស្វ័យប្រវត្ត (Convention Over Configuration)៖
  • Table Names៖ ឈ្មោះ Property របស់ DbSet (ឧ. Students) នឹងត្រូវបានយកទៅដាក់ជាឈ្មោះ Table នៅក្នុង SQL Server ដោយស្វ័យប្រវត្ត។
  • Primary Keys៖ Property នៅក្នុង Model ដែលមានឈ្មោះ Id ឬ StudentId នឹងត្រូវបានសម្គាល់ជា Primary Key (Identity) ដែលកើនឡើងលេខរៀងដោយខ្លួនឯង។
  • Data Types៖ ប្រភេទ string ក្នុង C# នឹងត្រូវបានបម្លែងទៅជា nvarchar(max) ហើយប្រភេទ int នឹងក្លាយជា int នៅក្នុង SQL។

Dependency Injection (DI) គឺជាបច្ចេកទេសរចនាកូដ (Design Pattern) ដែលត្រូវបានប្រើដើម្បីកាត់បន្ថយភាពចងភ្ជាប់គ្នា (Loose Coupling) រវាង Classes នីមួយៗ។ នៅក្នុង ASP.NET Core យើងមិនតម្រូវឱ្យបង្កើត Object នៃ DbContext ដោយប្រើពាក្យ new ឡើយ ផ្ទុយទៅវិញយើងឱ្យប្រព័ន្ធ DI Container ជាអ្នកចាត់ចែងបង្កើត និងបញ្ជូនវាទៅកាន់ Controller តាមរយៈ Constructor។
ប្រភេទអាយុកាលសេវាកម្ម (Service Lifetimes) ក្នង DI៖
ប្រភេទអាយុកាល ដំណើរការ និងការបង្កើត Object
Transient បង្កើត Object ថ្មីជានិច្ច រាល់ពេលដែលមានការស្នើសុំទាញយកទៅប្រើប្រាស់។
Scoped បង្កើត Object តែមួយគត់ សម្រាប់រាល់ដំណើរការនៃ HTTP Request នីមួយៗ។
(ចំណាំ៖ DbContext ត្រូវបានចុះឈ្មោះជាប្រភេទនេះជានិច្ច)
Singleton បង្កើត Object តែមួយគត់ តាំងពីកម្មវិធីចាប់ផ្តើមរហូតដល់កម្មវិធីបិទ (Shared for all requests)។
ការចុះឈ្មោះ AppDbContext នៅក្នុងឯកសារ Program.cs ៖
Program.cs
using Microsoft.EntityFrameworkCore;
using SchoolApp.Data;

var builder = WebApplication.CreateBuilder(args);

// ១. ចាប់យកខ្សែស្រឡាយតភ្ជាប់ពីឯកសារ appsettings.json
var connectionString = builder.Configuration.GetConnectionString("DefaultConnection");

// ២. ចុះឈ្មោះ AppDbContext ទៅក្នុង DI Container ដោយប្រាប់វាឱ្យប្រើ SQL Server
builder.Services.AddDbContext<AppDbContext>(options =>
    options.UseSqlServer(connectionString));

builder.Services.AddControllers();
var app = builder.Build();

app.UseAuthorization();
app.MapControllers();
app.Run();
